# Use official lightweight Python image
FROM python:3.11-slim

# Prevent Python from writing .pyc files and buffering output
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1

# Set working directory
WORKDIR /app

# Copy and install dependencies
COPY requirements.txt .

# Install system dependencies for pyautogui and other packages
RUN apt-get update && apt-get install -y \
    python3-tk \
    python3-dev \
    libx11-dev \
    libxtst-dev \
    libpng-dev \
    libjpeg-dev \
    && rm -rf /var/lib/apt/lists/*

# Install Python packages
RUN pip install --no-cache-dir -r requirements.txt gunicorn

# Copy project files into the container
COPY . .

# Expose the app port
EXPOSE 5000

# Set Flask environment variable (optional)
ENV FLASK_APP=app.py

# Run the Flask app with Gunicorn and Eventlet
# Replace 'app:app' with 'yourfilename:app' if your main Flask file isn't app.py
CMD ["gunicorn", "--worker-class", "eventlet", "-w", "1", "-b", "0.0.0.0:5000", "app:app"]
